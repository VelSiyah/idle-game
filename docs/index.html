import tkinter as tk class GameUI: def __init__(self, master, game): self.master = master self.game = game self.resource_label = tk.Label(master, text="Resources: 0") self.resource_label.pack() self.upgrade_button = tk.Button(master, text="Upgrade", command=self.upgrade) self.upgrade_button.pack() self.start_button = tk.Button(master, text="Start", command=self.start) self.start_button.pack() self.stop_button = tk.Button(master, text="Stop", command=self.stop) self.stop_button.pack() def update_ui(self): self.resource_label.config(text="Resources: {}".format(self.game.resource)) def upgrade(self): self.game.upgrade() self.update_ui() def start(self): self.game.start() def stop(self): self.game.stop() # Creating the Tkinter application root = tk.Tk() root.title("Idle Loops") # Creating an instance of the game UI game_ui = GameUI(root, game) def update(): game_ui.update_ui() root.after(1000, update) update() # Start the UI update loop root.mainloop()

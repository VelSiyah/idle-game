import tkinter as tk class GameUI: def __init__(self, player): self.player = player self.root = tk.Tk() self.resource_labels = {} for resource_name, resource in self.player.resources.items(): label = tk.Label(self.root, text=f"{resource_name}: {resource.amount}") label.pack() self.resource_labels[resource_name] = label produce_button = tk.Button(self.root, text="Produce", command=self.produce) produce_button.pack() def produce(self): self.player.produce() for resource_name, resource in self.player.resources.items(): self.resource_labels[resource_name].config(text=f"{resource_name}: {resource.amount}") self.root.after(1000, self.produce) # Example usage: player = Player() player.add_resource(Resource("Wood")) player.add_resource(Resource("Stone")) player.add_loop(Loop("Woodcutter", Resource("Wood"), [Resource("Stone", 5)])) ui = GameUI(player) ui.root.mainloop()
